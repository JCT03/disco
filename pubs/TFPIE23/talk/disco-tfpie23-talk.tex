\begin{frame}{\disco goals}
  \begin{itemize}
  \item Teach early CS students basic FP concepts
  \item Help students connect math and computation
  \item Enhance learning with an interactive platform
  \item Minimize notational \& conceptual friction
  \end{itemize}
\end{frame}

\begin{xframe}{Friction?}
  \begin{align*}
    f &: \N \to \Q \\
    f(2n)   &= 0 \\
    f(2n+1) &= \begin{cases} n/2 & \text{if } n > 5, \\
      n + 7 & \text{otherwise}
      \end{cases}
  \end{align*}

  \begin{overprint}
    \onslide<2>

    \begin{center}
  \begin{minted}{haskell}
f :: Int -> Rational
f x
  | even x    = 0
  | n > 5     = fromIntegral n / 2
  | otherwise = n + 7
  where
    n = x `div` 2
  \end{minted}
    \end{center}

  \onslide<3>
  \begin{minted}{text}
f : N -> Q
f(2n)   = 0
f(2n+1) =
  {? n/2     if n > 5,
     n + 7   otherwise
  ?}
  \end{minted}
  \end{overprint}
\end{xframe}
